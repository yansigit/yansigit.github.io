import cv2
import numpy as np

img = cv2.imread('image.jpg')

x=320; y=150; w=50; h=50;
roi = img[y:y+h, x:x+w] # y ~ y+y-1행, x ~ x+x-1 열 까지 슬라이싱
img2 = roi.copy()
img[y:y+h, x+w:x+w+w] = img2 # 바로 옆에다 관심영역 복사

cv2.imshow("img", img)
cv2.imshow("roi", roi)

cv2.waitKey(0)
cv2.destroyAllWindows()
