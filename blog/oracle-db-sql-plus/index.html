  1  select empno, ename, job, sal
  2  from emp
  3  where deptno = 10
  4* and job = 'CLERK'
SQL> del                             # 현재 행 삭제
SQL> l
  1  select empno, ename, job, sal
  2  from emp
  3* where deptno = 10

SQL> i and job = 'CLERK'
SQL> l
  1  select empno, ename, job, sal
  2  from emp
  3  where deptno = 10
  4* and job = 'CLERK'
SQL> del 3                           # 3행 삭제
SQL> l
  1  select empno, ename, job, sal
  2  from emp
  3* and job = 'CLERK'

SQL> del 2 3                         # 2~3 행 삭제
SQL> l
  1* select empno, ename, job, sal

  2  from emp
  3* where deptno = 20

  1  select empno, ename, deptno, sal
  2  from emp
  3* where sal >= 2000

SQL> del 1
SQL> l
  1  from emp
  2* where sal >= 2000
SQL> 0 select ename          # 첫번째 줄에 내용 삽입
SQL> l
  1  select ename
  2  from emp
  3* where sal >= 2000
